# Build stage
FROM golang:1.21-alpine AS builder
WORKDIR /app

# Install git for go modules
RUN apk add --no-cache git

COPY go.mod go.sum ./
RUN go mod download

COPY . .
RUN go build -o backend ./cmd/main.go

# Run stage
FROM alpine:latest
WORKDIR /root/
COPY --from=builder /app/backend .
COPY --from=builder /app/.env .  # if using .env

EXPOSE 8080
CMD ["./backend"]
